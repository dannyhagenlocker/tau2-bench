\begin{frame}{$\tau^3$-Bench: Two New Frontiers}
  \framesubtitle{\href{https://sierra.ai/blog/bench-advancing-agent-benchmarking-to-knowledge-and-voice}{sierra.ai/blog}}

  Since launching in June 2024, \taubench{} has become the \textbf{de facto standard} for evaluating whether AI agents can reliably complete tasks end-to-end.

  \vspace{0.8em}
  $\tau^3$-Bench expands the benchmark in two new directions:

  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{$\tau$-Knowledge}

      \vspace{0.3em}
      Can agents reason over \textbf{large, messy knowledge bases}?
      \begin{itemize}
        \item New Banking domain: 698 docs, $\sim$195K tokens
        \item Best model (GPT-5.2, high reasoning): \textbf{25\%}
        \item Even with gold documents: only 40\%
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{$\tau$-Voice}

      \vspace{0.3em}
      Can agents complete tasks over \textbf{realistic voice calls}?
      \begin{itemize}
        \item Full-duplex voice with noise, accents, interruptions
        \item 278 tasks across 3 domains
        \item Best voice agent retains only \textbf{30--45\%} of text capability
      \end{itemize}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{The Problem: Can Voice Agents Actually Do the Job?}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Full-duplex voice agents} listen and speak simultaneously---the next frontier in conversational AI.

      \vspace{1em}
      But existing benchmarks evaluate capabilities \textbf{in isolation}:
      \begin{itemize}
        \item \tautwobench{}: complex grounded tasks, but \textbf{text-only}
        \item Full-Duplex-Bench: turn-taking dynamics, but \textbf{no real tools}
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Voice compounds task difficulty:}
      \begin{itemize}
        \item No punctuation, fillers, disfluencies
        \item Accents, background noise, telephony compression
        \item Interruptions, backchannels, real-time turn-taking
      \end{itemize}

      \vspace{1em}
      \textcolor{SierraOrange}{\interbold{Missing:}} end-to-end evaluation of voice agents on \textbf{grounded tasks with real-world complexity}.
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Error Analysis \& Takeaways}

  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      \textbf{91 failed simulations analyzed:}

      \vspace{0.3em}
      \begin{small}
      \begin{tabular}{@{}lcc@{}}
        \toprule
        \textbf{Error Source} & \makecell{\textbf{Voice-}\\\textbf{Fragile}} & \makecell{\textbf{Noise-}\\\textbf{Fragile}} \\
        \midrule
        Agent errors & \textbf{79\%} & \textbf{90\%} \\
        \quad Logical & 13 & 16 \\
        \quad Transcription & 10 & 16 \\
        \quad Hallucination & 6 & 6 \\
        \midrule
        User sim errors & 21\% & 10\% \\
        \bottomrule
      \end{tabular}
      \end{small}

      \vspace{0.3em}
      {\small Failures reflect genuine agent limitations, not simulator artifacts.}
    \end{column}
    \begin{column}{0.53\textwidth}
      \textbf{Key takeaways:}

      \vspace{0.2em}
      \begin{enumerate}
        \item \textbf{The voice--text gap is real and large.} Best voice retains only 45\% of text SOTA---even under clean conditions.

        \item \textbf{It's not just ASR.} Logical errors and hallucinations dominate even with clean audio.

        \item \textbf{Acoustic realism matters unevenly.} Accents are devastating for some providers but not others---accessibility concerns.

        \item \textbf{No provider does it all.} Fundamental trade-off between responsiveness and restraint.
      \end{enumerate}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Failure Mode Details}
  \framesubtitle{Common patterns from qualitative error analysis}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Authentication bottleneck:}
      \begin{itemize}
        \item Agents fail to transcribe names/emails even when spelled letter-by-letter
        \item Blocks all downstream actions
        \item Compounds with accents
      \end{itemize}

      \vspace{0.5em}
      \textbf{Hallucinated completions:}
      \begin{itemize}
        \item Agent says ``I've updated your address'' without making any tool call
        \item Policy violations under time pressure
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Multi-step tracking failures:}
      \begin{itemize}
        \item Completes part of a request, forgets remaining items
        \item Information across turns gets lost
      \end{itemize}

      \vspace{0.5em}
      \textbf{Compounding under realistic conditions:}
      \begin{itemize}
        \item Unresponsive after repeated auth failures
        \item Verbosity causes timeouts on complex tasks
        \item Real-time pressure prevents ``thinking silently''
      \end{itemize}
    \end{column}
  \end{columns}

\end{frame}
